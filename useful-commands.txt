Some useful commands for set up and other stuff

step 1 - // setting up a virtual environment - only run once
python3 -m venv .venv

step 2 - // activating the environment
source .venv/bin/activate

step 3 - // install requirements - run once
pip install -r requirements.txt

// installing dependencies
pip install **

// running python files
python3 <file_name>

// deactivating environment
deactivate
